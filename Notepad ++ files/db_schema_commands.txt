RAGESCHEMA Setup script : 

CREATE USER &&rageschema IDENTIFIED BY &&rageschemapass DEFAULT TABLESPACE USERS TEMPORARY TABLESPACE &&TMPTBS;
ALTER USER &&rageschema ACCOUNT UNLOCK;
-- GRANTS
grant create session to &&rageschema;
grant connect to &&rageschema;
GRANT RESOURCE TO &&rageschema;
grant create view to &&rageschema;
grant create sequence to &&rageschema;
ALTER USER &&rageschema QUOTA UNLIMITED ON LAU_DATA;
ALTER USER &&rageschema QUOTA UNLIMITED ON LAU_INDEX;
ALTER USER &&rageschema QUOTA UNLIMITED ON USERS;

--create rageextschema account
CREATE USER &&rageextschema IDENTIFIED BY &&rageexpass DEFAULT TABLESPACE USERS TEMPORARY TABLESPACE &&TMPTBS;
ALTER USER &&rageextschema ACCOUNT UNLOCK;
-- GRANTS
grant create session to &&rageextschema;
ALTER USER &&rageextschema QUOTA UNLIMITED ON LAU_DATA;
ALTER USER &&rageextschema QUOTA UNLIMITED ON LAU_INDEX;
ALTER USER &&rageextschema QUOTA UNLIMITED ON USERS;
grant connect to &&rageextschema;
grant create session to &&rageextschema;
GRANT RESOURCE TO &&rageextschema ;
grant create table to &&rageextschema;
grant create sequence to &&rageextschema;

PVAIINTAKE Script : 

CREATE USER &&coreschema IDENTIFIED BY &&coreschemapass DEFAULT TABLESPACE &&TBS TEMPORARY TABLESPACE &&TMPTBS;
ALTER USER &&coreschema ACCOUNT UNLOCK;

-- grnat space quota

ALTER USER &&coreschema QUOTA UNLIMITED ON &&TBS;
ALTER USER &&coreschema QUOTA UNLIMITED ON &&INDXTBS;
ALTER USER &&coreschema QUOTA UNLIMITED ON &&LOBTBS;
ALTER USER &&coreschema QUOTA UNLIMITED ON USERS;


-- grant privilege

grant CREATE JOB to &&coreschema;
grant CREATE TABLE to &&coreschema;
grant CREATE ANY SEQUENCE to &&coreschema;
grant CREATE TYPE to &&coreschema;
grant ALTER ANY SEQUENCE to &&coreschema;
grant DEBUG CONNECT SESSION to &&coreschema;
grant DROP ANY SEQUENCE to &&coreschema;

grant CREATE TRIGGER to &&coreschema;
grant CREATE SESSION to &&coreschema;
grant CREATE PROCEDURE to &&coreschema;
